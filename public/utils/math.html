 96def angle(v1: Vector, v2: Vector) -> float:
 97    """
 98    Calculate the angle between two vectors in radians.
 99
100    Parameters
101    ----------
102    v1 : Vector
103        The first vector, represented as a list, tuple, or numpy ndarray of floats.
104    v2 : Vector
105        The second vector, represented as a list, tuple, or numpy ndarray of floats.
106
107    Returns
108    -------
109    float
110        The angle in radians between `v1` and `v2`, computed using the dot product formula.
111
112    Raises
113    ------
114    TypeError
115        If `v1` or `v2` is not a list, tuple, or numpy ndarray of floats.
116    ValueError
117        If `v1` and `v2` do not have the same length.
118
119    Notes
120    -----
121    - The angle is calculated using the arccosine of the normalized dot product of the vectors.
122    - Ensure that both vectors are non-zero before calling this function.
123    """
124    if not isinstance(v1, (list, tuple, np.ndarray)) or not isinstance(
125        v2, (list, tuple, np.ndarray)
126    ):
127        raise TypeError("Both v1 and v2 must be lists, tuples, or ndarrays of floats.")
128    if len(v1) != len(v2):
129        raise ValueError("Vectors v1 and v2 must be of the same length.")
130    angle = np.arccos(dotproduct(v1, v2) / (length(v1) * length(v2)))
131    return angle

246def calculate_rotation_between_vectors(
247    v_from: np.ndarray, v_to: np.ndarray
248) -> np.ndarray:
249    """
250    Calculate the rotation matrix that rotates one vector to another.
251
252    Parameters
253    ----------
254    v_from : np.ndarray
255        The initial vector.
256    v_to : np.ndarray
257        The target vector.
258
259    Returns
260    -------
261    np.ndarray
262        The rotation matrix.
263    """
264    v_from = normalize_vector(v_from)
265    v_to = normalize_vector(v_to)
266
267    angle = np.arccos(
268        np.dot(v_from, v_to) / (np.linalg.norm(v_from) * np.linalg.norm(v_to))
269    )
270    axis = np.cross(v_from, v_to)
271    axis_norm = np.linalg.norm(axis)
272
273    if axis_norm < 1e-5:
274        if angle < 1e-5:
275            return np.identity(3)
276        elif np.pi - angle < 1e-5:
277            axis = arbitrary_orthogonal_vector(v_from)
278            axis_norm = np.linalg.norm(axis)
279
280    axis /= axis_norm
281    r = R.from_rotvec(axis * angle)
282    return r.as_matrix()

492def random_unit_quaternion() -> UnitQuaternion:
493    """
494    Generate a random unit quaternion.
495
496    Returns
497    -------
498    UnitQuaternion
499        A random unit quaternion.
500    """
501    z = 100000
502    while z > 1.0:
503        x = random.uniform(-1.0, 1.0)
504        y = random.uniform(-1.0, 1.0)
505        z = x * x + y * y
506    w = 100000
507    while w > 1.0:
508        u = random.uniform(-1.0, 1.0)
509        v = random.uniform(-1.0, 1.0)
510        w = u * u + v * v
511
512    s = np.sqrt((1 - z) / w)
513    return UnitQuaternion([x, y, s * u, s * v])

601def quat_to_axang(q: np.ndarray) -> np.ndarray:
602    """
603    Convert a quaternion or an array of quaternions to a 4D axis-angle representation.
604
605    Parameters
606    ----------
607    q : np.ndarray
608        A numpy array of shape (4,) representing a single quaternion (w, x, y, z)
609        or of shape (N, 4) representing N quaternions.
610
611    Returns
612    -------
613    np.ndarray
614        A numpy array of shape (4,) representing the axis-angle of a single quaternion
615        (x, y, z, angle) or of shape (N, 4) representing the axis-angle representations
616        of N quaternions.
617    """
618    # Ensure the input is a 2D array for consistent processing
619    if q.ndim == 1:
620        q = np.expand_dims(q, axis=0)
621
622    # Normalize the quaternions
623    q_norm = q / np.linalg.norm(q, axis=1, keepdims=True)
624
625    # Compute the angle
626    angles = 2 * np.arccos(np.clip(q_norm[:, 0], -1.0, 1.0))
627
628    # Compute the axis, avoiding division by zero
629    s = np.sqrt(1 - q_norm[:, 0] ** 2)
630    axis = np.where(
631        s[:, np.newaxis] > 1e-5, q_norm[:, 1:] / s[:, np.newaxis], q_norm[:, 1:]
632    )
633
634    # Concatenate the axis and angle to form a 4D representation
635    axis_angle = np.hstack((axis, angles[:, np.newaxis]))
636
637    # If the original input was a single quaternion, return a single axis-angle
638    if axis_angle.shape[0] == 1:
639        return axis_angle[0]
640
641    return axis_angle

644def conj(q: Union[np.ndarray, list, sm.UnitQuaternion]) -> np.ndarray:
645    """
646    Compute the conjugate of an array of quaternions.
647
648    Parameters
649    ----------
650    q : Union[np.ndarray, list, sm.UnitQuaternion]
651        Input quaternion(s) which can be a list, numpy array, or sm.UnitQuaternion.
652
653    Returns
654    -------
655    np.ndarray
656        The conjugated array of quaternions.
657
658    Raises
659    ------
660    ValueError
661        If the input cannot be converted to a valid quaternion array.
662    """
663    # Convert sm.UnitQuaternion to numpy array
664    if isinstance(q, sm.UnitQuaternion):
665        q = np.array([q.s, *q.v])
666
667    # Convert list to numpy array
668    if isinstance(q, list):
669        q = np.array(q)
670
671    # Ensure the array is 1D or higher and has the correct shape
672    if q.ndim == 1:
673        if q.shape[0] != 4:
674            raise ValueError("1D quaternion must have 4 components (w, x, y, z).")
675        q_conj = np.copy(q)
676        q_conj[1:] *= -1
677    else:
678        if q.shape[-1] != 4:
679            raise ValueError("Quaternions must have 4 components (w, x, y, z).")
680        q_conj = np.copy(q)
681        q_conj[..., 1:] *= -1
682
683    return q_conj

801    transformed_points_T1 = np.dot(T1.A[:3, :3], points.T).T + T1.t
802    transformed_points_T2 = np.dot(T2.A[:3, :3], points.T).T + T2.t
803
804    dist_matrix = np.linalg.norm(
805        transformed_points_T1[:, np.newaxis, :]
806        - transformed_points_T2[np.newaxis, :, :],
807        axis=2,
808    )
809    hausdorff_dist = max(
810        np.max(np.min(dist_matrix, axis=1)), np.max(np.min(dist_matrix, axis=0))
811    )
812    return hausdorff_dist

815def cint(
816    F: Callable[[complex], complex],
817    C: Callable[[float], complex],
818    dC: Callable[[float], complex],
819    t_start: float = 0.0,
820    t_end: float = 2 * np.pi,
821    max_iter: int = 100,
822):
823    """
824    Computes the contour integral of a complex function along a parameterized curve.
825
826    This function performs numerical integration of a complex function F(z) along a curve C(t),
827    where t is the parameter along the curve, and dC(t) is the derivative of the curve with respect
828    to t. The integral is calculated from t_start to t_end using Gaussian quadrature.
829
830    Parameters
831    ----------
832    F : Callable[[complex], complex]
833        The complex function to integrate.
834    C : Callable[[float], complex]
835        A parameterized function representing the curve along which to integrate.
836    dC : Callable[[float], complex]
837        The derivative of the curve with respect to the parameter t.
838    t_start : float, optional
839        The starting value of the parameter t, by default 0.0.
840    t_end : float, optional
841        The ending value of the parameter t, by default 2 * np.pi.
842    max_iter : int, optional
843        The maximum number of iterations for the numerical integration, by default 100.
844
845    Returns
846    ----------
847    tuple
848        A tuple (I, I_e) where I is the result of the contour integral, and I_e is the estimated error.
849    """
850
851    def cintegral(
852        t,
853        F: Callable[[complex], complex],
854        C: Callable[[float], complex],
855        dC: Callable[[float], complex],
856    ):
857        """
858        Computes the complex integral at a specific parameter value along a curve.
859
860        This function evaluates the integrand at a specific parameter value `t`, which
861        corresponds to a point on a curve defined by the function `C(t)`. It computes
862        the product of the function `F(z)` at the point `z = C(t)` and the derivative
863        of the curve `dC(t)` with respect to `t`.
864
865        Parameters
866        ----------
867        t : float
868            The parameter value along the curve at which to evaluate the integral.
869        F : Callable[[complex], complex]
870            A complex-valued function to integrate.
871        C : Callable[[float], complex]
872            A parameterized curve as a function of `t` that returns a complex number `z`.
873        dC : Callable[[float], complex]
874            The derivative of the parameterized curve `C(t)` with respect to `t`.
875
876        Returns
877        ----------
878        complex
879            The product of `F(z)` and the derivative `dC(t)` at the point `z = C(t)`,
880            representing the integrand at `t`.
881        """
882        z = C(t)
883        dz_dt = dC(t)
884        return F(z) * dz_dt
885
886    integral_real, real_error = quad(
887        lambda t: np.real(cintegral(t, F, C, dC)),
888        t_start,
889        t_end,
890        complex_func=True,
891        limit=max_iter,
892    )
893    integral_imag, imag_error = quad(
894        lambda t: np.imag(cintegral(t, F, C, dC)),
895        t_start,
896        t_end,
897        complex_func=True,
898        limit=max_iter,
899    )
900    I = integral_real + 1j * integral_imag
901    I_e = real_error + 1j * imag_error
902    return I, I_e

905def homotopy_class(F: Callable[[complex], complex], points: np.ndarray):
906    """
907    Computes the homotopy class of a curve by integrating a complex function along the curve.
908
909    This function numerically evaluates the contour integral of a complex function F along a curve
910    defined by a set of 2D points. The integral result provides information about the homotopy class
911    of the curve with respect to the function F.
912
913    Parameters
914    ----------
915    F : Callable[[complex], complex]
916        The complex function to integrate along the curve.
917    points : np.ndarray
918        A NumPy array of shape (N, 2) representing the curve as a sequence of 2D points.
919
920    Returns
921    ----------
922    tuple
923        A tuple (I, I_e) where I is the result of the contour integral, and I_e is the estimated error.
924    """
925
926    t_start = 0
927    t_end = 1
928
929    def C(t: float) -> complex:
930        """
931        Returns a complex number representing a point on a piecewise linear curve.
932
933        This function computes the position of a point on a piecewise linear curve
934        parameterized by `t`. The curve is defined by a sequence of points in a
935        2D plane, and `t` is used to interpolate between these points. If `t` is
936        less than or equal to 0, the function returns the first point on the curve.
937        If `t` is greater than or equal to 1, the function returns the last point.
938
939        Parameters
940        ----------
941        t : float
942            A parameter value between 0 and 1 representing a point along the curve.
943            Values less than 0 return the first point, and values greater than 1
944            return the last point.
945
946        Returns
947        ----------
948        complex
949            A complex number representing the interpolated point on the curve, where
950            the real part corresponds to the x-coordinate and the imaginary part
951            corresponds to the y-coordinate.
952        """
953        if t <= 0:
954            return complex(real=points[0][0], imag=points[0][1])
955        elif t >= 1:
956            return complex(real=points[-1][0], imag=points[-1][1])
957
958        segment_index = int(t * (len(points) - 1))
959        p0 = points[segment_index]
960        p1 = points[segment_index + 1]
961        t_local = (t * (len(points) - 1)) - segment_index
962
963        x = p0[0] + (p1[0] - p0[0]) * t_local
964        y = p0[1] + (p1[1] - p0[1]) * t_local
965
966        return complex(real=x, imag=y)
967
968    def dC(t: float, delta_t: float = 1e-6) -> complex:
969        """
970        Computes the derivative of a parameterized curve at a given point.
971
972        This function calculates the numerical derivative of the parameterized
973        curve `C(t)` at a specific point `t` using central difference approximation.
974        The derivative represents the rate of change of the curve with respect to
975        `t`, returning a complex number where the real part represents the x-component
976        of the derivative and the imaginary part represents the y-component.
977
978        Parameters
979        ----------
980        t : float
981            The parameter value at which to compute the derivative along the curve.
982        delta_t : float, optional
983            The step size used for central difference approximation (default is 1e-6).
984
985        Returns
986        ----------
987        complex
988            A complex number representing the derivative of the curve at `t`, where
989            the real part corresponds to the derivative of the x-coordinate, and the
990            imaginary part corresponds to the derivative of the y-coordinate.
991        """
992        next_point = C(t + delta_t)
993        prev_point = C(t - delta_t)
994        dz_dt = (next_point - prev_point) / (2 * delta_t)
995        return dz_dt
996
997    I, I_e = cint(F, C, dC, t_start, t_end)
998    return I, I_e

907    """
908    Computes the homotopy class of a curve by integrating a complex function along the curve.
909
910    This function numerically evaluates the contour integral of a complex function F along a curve
911    defined by a set of 2D points. The integral result provides information about the homotopy class
912    of the curve with respect to the function F.
913
914    Parameters
915    ----------
916    F : Callable[[complex], complex]
917        The complex function to integrate along the curve.
918    points : np.ndarray
919        A NumPy array of shape (N, 2) representing the curve as a sequence of 2D points.
920
921    Returns
922    ----------
923    tuple
924        A tuple (I, I_e) where I is the result of the contour integral, and I_e is the estimated error.
925    """
926
927    t_start = 0
928    t_end = 1
929
930    def C(t: float) -> complex:
931        """
932        Returns a complex number representing a point on a piecewise linear curve.
933
934        This function computes the position of a point on a piecewise linear curve
935        parameterized by `t`. The curve is defined by a sequence of points in a
936        2D plane, and `t` is used to interpolate between these points. If `t` is
937        less than or equal to 0, the function returns the first point on the curve.
938        If `t` is greater than or equal to 1, the function returns the last point.
939
940        Parameters
941        ----------
942        t : float
943            A parameter value between 0 and 1 representing a point along the curve.
944            Values less than 0 return the first point, and values greater than 1
945            return the last point.
946
947        Returns
948        ----------
949        complex
950            A complex number representing the interpolated point on the curve, where
951            the real part corresponds to the x-coordinate and the imaginary part
952            corresponds to the y-coordinate.
953        """
954        if t <= 0:
955            return complex(real=points[0][0], imag=points[0][1])
956        elif t >= 1:
957            return complex(real=points[-1][0], imag=points[-1][1])
958
959        segment_index = int(t * (len(points) - 1))
960        p0 = points[segment_index]
961        p1 = points[segment_index + 1]
962        t_local = (t * (len(points) - 1)) - segment_index
963
964        x = p0[0] + (p1[0] - p0[0]) * t_local
965        y = p0[1] + (p1[1] - p0[1]) * t_local
966
967        return complex(real=x, imag=y)
968
969    def dC(t: float, delta_t: float = 1e-6) -> complex:
970        """
971        Computes the derivative of a parameterized curve at a given point.
972
973        This function calculates the numerical derivative of the parameterized
974        curve `C(t)` at a specific point `t` using central difference approximation.
975        The derivative represents the rate of change of the curve with respect to
976        `t`, returning a complex number where the real part represents the x-component
977        of the derivative and the imaginary part represents the y-component.
978
979        Parameters
980        ----------
981        t : float

983        delta_t : float, optional

985
986        Returns
987        ----------

990            the real part corresponds to the derivative of the x-coordinate, and the
991            imaginary part corresponds to the derivative of the y-coordinate.
992        """
993        next_point = C(t + delta_t)
994        prev_point = C(t - delta_t)
995        dz_dt = (next_point - prev_point) / (2 * delta_t)
996        return dz_dt
