 22    def __init__(
 23        self,
 24        n_bfs: int = 10,
 25        alpha: float = 48.0,
 26        beta: Optional[float] = None,
 27        cs_alpha: Optional[float] = None,
 28        cs: Optional[CanonicalSystem] = None,
 29        roto_dilatation: bool = False,
 30    ):
 31        """
 32        Initialize the DMPQuaternion.
 33
 34        Parameters
 35        ----------
 36        n_bfs : int
 37            Number of basis functions.
 38        alpha : float
 39            Filter constant.
 40        beta : Optional[float]
 41            Filter constant. If None, set to alpha / 4.
 42        cs_alpha : Optional[float]
 43            Alpha value for the canonical system. If None, set to alpha / 2.
 44        cs : Optional[CanonicalSystem]
 45            Canonical system instance. If None, create a new one with cs_alpha.
 46        roto_dilatation : bool
 47            Flag to enable roto-dilatation.
 48        """
 49        self.n_bfs = n_bfs
 50        self.alpha = alpha
 51        self.beta = beta if beta is not None else self.alpha / 4
 52        self.cs = (
 53            cs
 54            if cs is not None
 55            else CanonicalSystem(
 56                alpha=cs_alpha if cs_alpha is not None else self.alpha / 2
 57            )
 58        )
 59
 60        # Centres of the Gaussian basis functions
 61        self.c = np.exp(-self.cs.alpha * np.linspace(0, 1, self.n_bfs))
 62
 63        # Variance of the Gaussian basis functions
 64        self.h = 1.0 / np.gradient(self.c) ** 2
 65
 66        # Scaling factor
 67        self.Do = np.identity(3)
 68
 69        # Initially weights are zero (no forcing term)
 70        self.w = np.zeros((3, self.n_bfs))
 71
 72        # Initial and goal orientations
 73        self._q0 = quaternion.one
 74        self._go = quaternion.one
 75
 76        self._q0_train = quaternion.one
 77        self._go_train = quaternion.one
 78
 79        self._R_fx = np.identity(3)
 80
 81        # Reset
 82        self.q = self._q0.copy()
 83        self.omega = np.zeros(3)
 84        self.d_omega = np.zeros(3)
 85        self.train_quats = None
 86        self.train_omega = None
 87        self.train_d_omega = None
 88
 89        self._roto_dilatation = roto_dilatation

 91    def step(
 92        self,
 93        x: float,
 94        dt: float,
 95        tau: float,
 96        torque_disturbance: np.ndarray = np.array([0, 0, 0]),
 97    ) -> Tuple[quaternion.quaternion, np.ndarray, np.ndarray]:
 98        """
 99        Perform a single DMP step.
100
101        Parameters
102        ----------
103        x : float
104            Phase variable.
105        dt : float
106            Time step.
107        tau : float
108            Temporal scaling factor.
109        torque_disturbance : np.ndarray, optional
110            External torque disturbance, by default np.array([0, 0, 0]).
111
112        Returns
113        -------
114        Tuple[quaternion.quaternion, np.ndarray, np.ndarray]
115            Current quaternion, angular velocity, and angular acceleration.
116        """
117
118        def fo(xj):
119            psi = np.exp(-self.h * (xj - self.c) ** 2)
120            return self.Do.dot(self.w.dot(psi) / psi.sum() * xj)
121
122        # DMP system acceleration
123        self.d_omega = (
124            self.alpha
125            * (
126                self.beta * 2 * np.log(self._go * self.q.conjugate()).vec
127                - tau * self.omega
128            )
129            + self._R_fx @ fo(x)
130            + torque_disturbance
131        ) / tau**2
132
133        # Integrate rotational acceleration
134        self.omega += self.d_omega * dt
135
136        # Integrate rotational velocity (to obtain quaternion)
137        self.q = np.exp(dt / 2 * np.quaternion(0, *self.omega)) * self.q
138
139        return npq2np(self.q), self.omega, self.d_omega

141    def rollout(
142        self, ts: np.ndarray, tau: Union[float, np.ndarray]
143    ) -> Tuple[np.ndarray, np.ndarray, np.ndarray]:
144        """
145        Perform a DMP rollout over the given time steps.
146
147        Parameters
148        ----------
149        ts : np.ndarray
150            Array of time steps.
151        tau : Union[float, np.ndarray]
152            Temporal scaling factor.
153
154        Returns
155        -------
156        Tuple[np.ndarray, np.ndarray, np.ndarray]
157            Arrays of quaternions, angular velocities, and angular accelerations.
158        """
159        self.reset()
160
161        if np.isscalar(tau):
162            tau = np.full_like(ts, tau)
163
164        x = self.cs.rollout(ts, tau)  # Integrate canonical system
165        dt = np.gradient(ts)  # Differential time vector
166
167        n_steps = len(ts)
168        q = np.empty((n_steps,), dtype=np.quaternion)
169        omega = np.empty((n_steps, 3))
170        d_omega = np.empty((n_steps, 3))
171
172        for i in range(n_steps):
173            q[i], omega[i], d_omega[i] = self.step(x[i], dt[i], tau[i])
174
175        return q, omega, d_omega

185    def train(self, quaternions: np.ndarray, ts: np.ndarray, tau: float) -> None:
186        """
187        Train the DMP with given quaternion trajectory.
188
189        Parameters
190        ----------
191        quaternions : np.ndarray
192            Array of quaternions.
193        ts : np.ndarray
194            Array of time steps.
195        tau : float
196            Temporal scaling factor.
197
198        Raises
199        ------
200        RuntimeError
201            If the length of quaternions does not match the length of time steps.
202        """
203        # View input as numpy quaternion type
204        if quaternions.dtype == np.quaternion:
205            quats = quaternions
206        else:
207            quats = quaternion.as_quat_array(quaternions)
208
209        # Sanity-check input
210        if len(quats) != len(ts):
211            raise RuntimeError("len(quats) != len(ts)")
212
213        # Initial and goal orientations
214        self._q0 = quats[0]
215        self._go = quats[-1]
216
217        self._q0_train = quats[0]
218        self._go_train = quats[-1]
219
220        # Differential time vector
221        dt = np.gradient(ts)[:, np.newaxis]
222
223        # Scaling factor
224        Do_inv = np.identity(3)
225
226        # Compute finite difference velocity between orientations
227        omega = 2 * np.log(np.roll(quats, -1) * quats.conjugate())  # In unit time
228        omega[-1] = omega[-2]  # Last element is no good
229        omega = quaternion.as_float_array(omega)[:, 1:] / dt  # Scale by dt
230
231        # Compute desired angular accelerations
232        d_omega = np.gradient(omega, axis=0) / dt
233
234        # Integrate canonical system at time points
235        x = self.cs.rollout(ts, tau)
236
237        # Set up system of equations to solve for weights
238        def features(xj):
239            psi = np.exp(-self.h * (xj - self.c) ** 2)
240            return xj * psi / psi.sum()
241
242        def forcing(j):
243            return Do_inv.dot(
244                tau**2 * d_omega[j]
245                - self.alpha
246                * (
247                    self.beta * (2 * np.log(self._go * quats[j].conjugate())).vec
248                    - tau * omega[j]
249                )
250            )
251
252        A = np.stack([features(xj) for xj in x])
253        f = np.stack([forcing(j) for j in range(len(ts))])
254
255        # Least squares solution for Aw = f (for each column of f)
256        self.w = np.linalg.lstsq(A, f, rcond=None)[0].T
257
258        # Cache variables for later inspection
259        self.train_quats = quats
260        self.train_omega = omega
261        self.train_d_omega = d_omega

296    def _update_goal_change_parameters(self) -> None:
297        """
298        Update parameters when the goal quaternion is changed.
299        """
300        self._sg = (
301            np.log(self._go_train * self._q0_train.conjugate()).norm()
302            / np.log(self._go * self._q0.conjugate()).norm()
303        )
304
305        v_new = (2 * np.log(self._go * self._q0.conjugate())).vec
306        v_train = (2 * np.log(self._go_train * self._q0_train.conjugate())).vec
307        self._R_fx = calculate_rotation_between_vectors(v_train, v_new)

186    def train(self, quaternions: np.ndarray, ts: np.ndarray, tau: float) -> None:
187        """
188        Train the DMP with given quaternion trajectory.
189
190        Parameters
191        ----------
192        quaternions : np.ndarray
193            Array of quaternions.
194        ts : np.ndarray
195            Array of time steps.
196        tau : float
197            Temporal scaling factor.
198
199        Raises
200        ------
201        RuntimeError
202            If the length of quaternions does not match the length of time steps.
203        """
204        # View input as numpy quaternion type
205        if quaternions.dtype == np.quaternion:
206            quats = quaternions
207        else:
208            quats = quaternion.as_quat_array(quaternions)
209
210        # Sanity-check input
211        if len(quats) != len(ts):
212            raise RuntimeError("len(quats) != len(ts)")
213
214        # Initial and goal orientations
215        self._q0 = quats[0]
216        self._go = quats[-1]
217
218        self._q0_train = quats[0]
219        self._go_train = quats[-1]
220
221        # Differential time vector
222        dt = np.gradient(ts)[:, np.newaxis]
223
224        # Scaling factor
225        Do_inv = np.identity(3)
226
227        # Compute finite difference velocity between orientations
228        omega = 2 * np.log(np.roll(quats, -1) * quats.conjugate())  # In unit time
229        omega[-1] = omega[-2]  # Last element is no good
230        omega = quaternion.as_float_array(omega)[:, 1:] / dt  # Scale by dt
231
232        # Compute desired angular accelerations
233        d_omega = np.gradient(omega, axis=0) / dt
234
235        # Integrate canonical system at time points
236        x = self.cs.rollout(ts, tau)
237
238        # Set up system of equations to solve for weights
239        def features(xj):
240            psi = np.exp(-self.h * (xj - self.c) ** 2)
241            return xj * psi / psi.sum()
242
243        def forcing(j):
244            return Do_inv.dot(
245                tau**2 * d_omega[j]
246                - self.alpha
247                * (
248                    self.beta * (2 * np.log(self._go * quats[j].conjugate())).vec
249                    - tau * omega[j]
250                )
251            )
252
253        A = np.stack([features(xj) for xj in x])
254        f = np.stack([forcing(j) for j in range(len(ts))])
255
256        # Least squares solution for Aw = f (for each column of f)
257        self.w = np.linalg.lstsq(A, f, rcond=None)[0].T
258
259        # Cache variables for later inspection
260        self.train_quats = quats
261        self.train_omega = omega
262        self.train_d_omega = d_omega
